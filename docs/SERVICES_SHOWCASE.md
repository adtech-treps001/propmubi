  const [services, setServices] = useState([]);

  useEffect(() => {
    // Fetch available services for this property

    setServices(availableServices);

  return (
    <View>
      <ServiceTabs>
        {services.map(service => (
          <ServiceCard
            key={service.id}
            name={service.name}
            cost={service.cost}
            onBook={() => bookService(propertyId, service.id)}
          />
        ))}
      </ServiceTabs>
    </View>
  );
};

app.post('/api/properties/:id/book-service', async (req, res) => {
  const { propertyId, serviceId } = req.params;
  const { userId, serviceType } = req.body;

  // Orchestrate workflow based on service type
  const result = await orchestrator.executeServiceWorkflow({
    propertyId,
    userId,
    serviceType
  });

  res.json(result);
